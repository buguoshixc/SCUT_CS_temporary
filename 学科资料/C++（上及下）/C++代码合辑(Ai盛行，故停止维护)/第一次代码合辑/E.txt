#include <iostream>
#include <string>

using namespace std;

class Student {
private:
    int id;
    string name;
    double hwGrade;
    double midtermGrade;
    double finalGrade;

public:
    Student(int _id, string _name, double _hwGrade, double _midtermGrade, double _finalGrade)
        : id(_id), name(_name), hwGrade(_hwGrade), midtermGrade(_midtermGrade), finalGrade(_finalGrade) {}

    // 重载乘法运算符，用于计算加权成绩
    void operator*() {
        hwGrade *= 0.2;
        midtermGrade *= 0.3;
        finalGrade *= 0.5;
    }

    // 重载输出运算符，用于输出成绩单
    friend ostream& operator<<(ostream& os, const Student& student) {
        os << "学生姓名: " << student.name << endl;
        os << "学生学号: " << student.id << endl;
        os << "平时作业成绩加权: " << student.hwGrade << endl;
        os << "期中考试成绩加权: " << student.midtermGrade << endl;
        os << "期末考试成绩加权: " << student.finalGrade << endl;
        os << "总成绩: " << (student.hwGrade + student.midtermGrade + student.finalGrade) << endl;
        return os;
    }
};

int main() {
    int id;
    string name;
    double hw, midterm, final;
    char continue_input = 'y';

    while (continue_input == 'y') {
        cout << "请输入学生学号: ";
        cin >> id;
        cout << "请输入学生姓名: ";
        cin >> name;
        cout << "请输入平时作业成绩: ";
        cin >> hw;
        cout << "请输入期中考试成绩: ";
        cin >> midterm;
        cout << "请输入期末考试成绩: ";
        cin >> final;

        Student student(id, name, hw, midterm, final);

        // 调用重载的乘法运算符进行加权成绩的计算
        student.operator*();

        cout << "学生信息及成绩单：" << endl;
        cout << student << endl;

        cout << "是否继续输入学生信息？(y/n): ";
        cin >> continue_input;
    }

    return 0;
}
