#include <iostream>
#include <string>

using namespace std;

class Student {
protected:
    int id;
    string name;
    double practiceCredits;
    double courseCredits;

public:
    Student(int _id, string _name, double _practiceCredits, double _courseCredits) : id(_id), name(_name), practiceCredits(_practiceCredits), courseCredits(_courseCredits) {}

    virtual void calculateCredits() = 0;

    virtual void getTotalCredit() {
        cout << "总学分：" << practiceCredits + courseCredits << endl;
    }

    virtual void disp() {
        cout << "学号：" << id << "，姓名：" << name << endl;
        getTotalCredit();
    }
};

class SchoolStuLeader : public Student {
public:
    SchoolStuLeader(int _id, string _name, double _days, double _course) : Student(_id, _name, 0, _course) {
        practiceCredits = (_days * 0.01 + 2) / 5;
    }

    void calculateCredits() override {}
};

class FacultiesStuLeader : public Student {
public:
    FacultiesStuLeader(int _id, string _name, double _days, double _course) : Student(_id, _name, 0, _course) {
        practiceCredits = (_days * 0.01 + 1) / 5;
    }

    void calculateCredits() override {}
};

class ClassStuLeader : public Student {
public:
    ClassStuLeader(int _id, string _name, double _days, double _course) : Student(_id, _name, 0, _course) {
        practiceCredits = (_days * 0.01 + 0.8) / 5;
    }

    void calculateCredits() override {}
};

class RegStudent : public Student {
public:
    RegStudent(int _id, string _name, double _hours, double _course) : Student(_id, _name, _hours * 0.005, _course) {}

    void calculateCredits() override {}
};

int main() {
    Student* student;

    cout << "请输入校级学生干部的信息，包括学号，名字，任职天数，课程学分：" << endl;
    int id;
    string name;
    double days;
    double course;
    cin >> id >> name >> days >> course;
    student = new SchoolStuLeader(id, name, days, course);
    student->calculateCredits();
    student->disp();

    cout << "请输入院级学生干部的信息，包括学号，名字，任职天数，课程学分：" << endl;
    cin >> id >> name >> days >> course;
    student = new FacultiesStuLeader(id, name, days, course);
    student->calculateCredits();
    student->disp();

    cout << "请输入班级学生干部的信息，包括学号，名字，任职天数，课程学分：" << endl;
    cin >> id >> name >> days >> course;
    student = new ClassStuLeader(id, name, days, course);
    student->calculateCredits();
    student->disp();

    cout << "请输入非学生干部的信息，包括学号，名字，课外实践时长，课程学分：" << endl;
    double hours;
    cin >> id >> name >> hours >> course;

    student = new RegStudent(id, name, hours, course);
    student->calculateCredits();
    student->disp();

    return 0;
}
