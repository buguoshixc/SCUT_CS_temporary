#include<iostream>
#include<string>
using namespace std;
class Student
{
private:
	int id;//以后都用int！
	string name;
	double hw;
	double midterm;
	double final;
	static double couscore1;

public:
	Student();
	Student(int a, string b,double c,double d,double e):id(a),name(b),hw(c),midterm(d),final(e){}
    void couscore()
    {
        couscore1 = hw * 0.2 + midterm * 0.3 + final * 0.5;
        cout << "总成绩: " << couscore1 << endl;
    }
    void disp()const
    {
        cout << "学生姓名: " << name << endl;
        cout << "学生学号: " << id << endl;
        cout << "平时作业成绩: " << hw << endl;
        cout << "期中考试成绩: " << midterm << endl;
        cout << "期末考试成绩: " << final << endl<<endl;
    }
};
double Student::couscore1= 0.0;
int main() {
    int id;
    string name;
    double hw, midterm, final;
    char continue_input = 'y'; // 控制是否继续输入的标志

    while (continue_input == 'y') {
        cout << "请输入学生学号: ";
        cin >> id;
        cout << "请输入学生姓名: ";
        cin >> name;
        cout << "请输入平时作业成绩: ";
        cin >> hw;
        cout << "请输入期中考试成绩: ";
        cin >> midterm;
        cout << "请输入期末考试成绩: ";
        cin >> final;

        Student student(id, name, hw, midterm, final);
        student.couscore();
        student.disp();

        cout << "是否继续输入学生信息？(y/n): ";
        cin >> continue_input;
    }

    return 0;
}
