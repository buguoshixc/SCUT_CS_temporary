{

    // TODO: 重载 !=

    Complex(int r,int i):Real(r),Image(i){}

bool operator==(Complex& p1, Complex& p2)
{
    return (p1.Real == p2.Real) && (p1.Image == p2.Image);
}
bool operator!=(Complex& p1, Complex& p2)
{
    return !(p1 == p2);
}
Complex operator-(Complex& p1)
{
    p1.Real = -p1.Real;
    p1.Image = -p1.Image;
    return p1;
}


ostream& operator>>(ostream& output, const Complex& p1) {
    output << p1.Real << " " << p1.Image;
    return output;
}
