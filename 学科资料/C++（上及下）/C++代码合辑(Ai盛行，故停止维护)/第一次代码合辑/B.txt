#include <iostream>
#include <string>
using namespace std;

class Film {
protected:
    string title;
    string director;
    int time;
    int quality;

public:
    virtual void input() {
        cout << "Input title:" << endl;
        getline(cin, title);
        cout << "Input director:" << endl;
        getline(cin, director);                                 //getline(cin,  )的使用
        cout << "Input time:" << endl;
        cin >> time;
        cout << "Input quality:" << endl;
        cin >> quality;
        cin.ignore(); // Clear newline character from input buffer
    }

    virtual void output() const {
        cout << "Title: " << title << endl;
        cout << "Director: " << director << endl;
        cout << "Time: " << time << " mins" << endl;
        cout << "Quality: ";
        for (int i = 0; i < quality; ++i) {
            cout << "*";
        }
        cout << endl;
    }
};

class DirectorCut : public Film {
private:
    int rev_time;
    string changes;

public:
    void input() override {                                                  //这个override注意一下啥意思
        Film::input();
        cout << "Input rev_time:" << endl;
        cin >> rev_time;
        cin.ignore(); // Clear newline character from input buffer
        cout << "Input changes:" << endl;
        getline(cin, changes);
    }

    void output() const override {
        Film::output();
        cout << "Revised time: " << rev_time << " mins" << endl;
        cout << "Changes: " << changes << endl;
    }
};

class ForeignFilm : public Film {
private:
    string language;

public:
    void input() override {
        Film::input();
        cout << "Input language:" << endl;
        getline(cin, language);
    }

    void output() const override {
        Film::output();
        cout << "Language: " << language << endl;
    }
};

Film* read_input(const string& class_name) {  //这里很重要！！
    Film* film = nullptr;
    if (class_name == "Film") {
        film = new Film();
    }
    else if (class_name == "DirectorCut") {
        film = new DirectorCut();
    }
    else if (class_name == "ForeignFilm") {
        film = new ForeignFilm();
    }
    if (film) {
        film->input();
    }
    return film;
}

int main() { 
   
    int n = 5;
    Film* films[5];
    for (int i = 0; i < n; i++) {
        cout << "Input class name:" << endl;
        string class_name;
        cin >> class_name;
        cin.ignore(); //可以去除输入流中的换行符
        films[i] = read_input(class_name);
        if (films[i]) {
            cout << class_name << "--" << endl;
            films[i]->output();
            cout << endl;
        }
    }

    // Clean up dynamically allocated memory
    for (int i = 0; i < n; i++) {
        delete films[i];
    }

    return 0;
}
